Based on discovery, create screenshot_agent_bridge.js that:

1. Exports `async function runScreenshot(url, chatId)`
2. Spawns `python3 /var/lib/openclaw/workspace/tools/screenshot_tool.py <url> <chatId>` as child process
3. Collects stdout, parses JSON result
4. Returns the parsed result object (or {ok: false, error: "..."} on spawn failure)
5. Sets timeout of 120 seconds (covers 30s capture + 60s bypass + buffer)

```javascript
const { spawn } = require('child_process');

const TOOL_PATH = '/var/lib/openclaw/workspace/tools/screenshot_tool.py';
const TIMEOUT_MS = 120_000;

async function runScreenshot(url, chatId) {
  return new Promise((resolve) => {
    const args = chatId ? [TOOL_PATH, url, chatId] : [TOOL_PATH, url];
    const proc = spawn('python3', args, { timeout: TIMEOUT_MS });

    let stdout = '';
    let stderr = '';
    proc.stdout.on('data', (d) => { stdout += d; });
    proc.stderr.on('data', (d) => { stderr += d; });

    proc.on('close', (code) => {
      try {
        resolve(JSON.parse(stdout.trim()));
      } catch (e) {
        resolve({ ok: false, error: `Tool output parse failed: ${stderr.slice(0, 200)}` });
      }
    });

    proc.on('error', (e) => {
      resolve({ ok: false, error: `Spawn failed: ${e.message}` });
    });
  });
}

module.exports = { runScreenshot };
```

```python
async def enroll_staging(url: str, cls: str, conf: float) -> None:
    """Write non-PASS URL to site_test_catalog with staging=true flag."""
    db_url = os.environ.get("DATABASE_URL", "")
    if not db_url:
        return  # no DB configured, skip silently
    try:
        conn = await asyncpg.connect(db_url)
        try:
            await conn.execute(
                """
                INSERT INTO site_test_catalog (url, staging, last_classification, confidence, enrolled_at)
                VALUES ($1, true, $2, $3, NOW())
                ON CONFLICT (url) DO UPDATE
                  SET staging = true,
                      last_classification = EXCLUDED.last_classification,
                      confidence = EXCLUDED.confidence,
                      enrolled_at = EXCLUDED.enrolled_at
                WHERE site_test_catalog.staging = false OR site_test_catalog.last_classification != EXCLUDED.last_classification
                """,
                url, cls, conf
            )
        finally:
            await conn.close()
    except Exception as e:
        # Non-fatal: log to stderr only
        print(f"[staging enroll error] {e}", file=sys.stderr)
```
