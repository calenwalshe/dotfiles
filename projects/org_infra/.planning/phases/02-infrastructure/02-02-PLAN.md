Implement a filesystem-based artifact store:

```python
class ArtifactStore:
    def __init__(self, base_dir: str | Path):
        """Initialize store with base directory for artifact persistence."""

    def write(self, run_id: str, agent_id: str, artifact: BaseModel) -> str:
        """Write artifact to store. Returns artifact path.
        Path: {base_dir}/{run_id}/{agent_id}.json
        Serializes via artifact.model_dump_json()"""

    def read(self, run_id: str, agent_id: str, model_class: type[T]) -> T:
        """Read artifact from store. Deserializes and validates against model_class.
        Raises FileNotFoundError if not found."""

    def list_artifacts(self, run_id: str) -> list[str]:
        """List all agent_ids that have artifacts for this run."""

    def exists(self, run_id: str, agent_id: str) -> bool:
        """Check if artifact exists without reading it."""
```
